<!DOCTYPE html>

<html lang="en">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <title>Random Number Generator</title>

    <script>

        function generateRandomNumber() {

            const now = Date.now();

            const lastGenerated = localStorage.getItem("lastGeneratedTime");

            const thirtyMinutes = 30 * 60 * 1000; // 30 minutes in milliseconds


            if (lastGenerated && now - lastGenerated < thirtyMinutes) {

                const savedNumber = localStorage.getItem("randomNumber");

                document.getElementById("result").innerText = "Your last number: " + savedNumber + " (Wait 30 minutes for a new one)";

                return;

            }


            const randomNumber = Math.floor(Math.random() * 100) + 1;

            localStorage.setItem("randomNumber", randomNumber);

            localStorage.setItem("lastGeneratedTime", now);

            document.getElementById("result").innerText = "Your Number: " + randomNumber;

        }


        window.onload = generateRandomNumber;

    </script>

</head>

<body>

    <h1>Random Number Generator</h1>

    <p id="result">Generating...</p>

</body>
