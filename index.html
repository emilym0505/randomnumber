    <script>

        function generateRandomNumber() {

            let lastScanTime = localStorage.getItem("lastScanTime");

            let now = new Date().getTime();

            let tenMinutes = 10 * 60 * 1000;

            

            if (!lastScanTime || now - lastScanTime >= tenMinutes) {

                let randomNum = Math.floor(Math.random() * 100) + 1;

                sessionStorage.setItem("randomNumber", randomNum);

                localStorage.setItem("lastScanTime", now);

                document.getElementById("random-number").innerText = randomNum;

                document.getElementById("message").innerText = "";

            } else {

                document.getElementById("random-number").innerText = "-";

                document.getElementById("message").innerText = "You can only scan once every 10 minutes. Please try again later.";

            }

        }

        

        window.onload = function() {

            generateRandomNumber();

        };

    </script>
